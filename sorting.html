Algorithm

Start
Input n; fact ‚Üê 1

Output fact
End

Flowchart

Start ‚Üí Read n ‚Üí fact = 1 ‚Üí i = 1 ‚Üí i ‚â§ n? ‚Üí fact = fact * i ‚Üí i = i + 1 ‚Üí Repeat ‚Üí Print factorial ‚Üí End

import java.util.Scanner;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        if(n < 0){ 
            System.out.println("Not defined for negatives.");

        for(int i = 1; i <= n; i++) fact *= i;
        System.out.println("Factorial of " + n + " is " + fact);
    }
}

8) Fibonacci Series

Algorithm

START
INPUT n
a ‚Üê 0, b ‚Üê 1
REPEAT n times:
  PRINT a
  next ‚Üê a + b
  a ‚Üê b
  b ‚Üê next
END

Flowchart

Start ‚Üí Read n ‚Üí a=0, b=1 ‚Üí Print a ‚Üí Print b ‚Üí i=3 ‚Üí i ‚â§ n? ‚Üí c = a + b ‚Üí Print c ‚Üí a = b ‚Üí b = c ‚Üí i = i + 1 ‚Üí Repeat ‚Üí End

Java Program

import java.util.Scanner;
class Fibonacci {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter n: ");
        int n = sc.nextInt();
        int a = 0, b = 1;
        if(n >= 1) System.out.print(a + " ");
        if(n >= 2) System.out.print(b + " ");
        for(int i = 3; i <= n; i++){
            int c = a + b;
            System.out.print(c + " ");
            a = b;
            b = c;
        }
    }
}

Pseudocode

START
READ n
a ‚Üê 0
b ‚Üê 1
PRINT a
PRINT b
FOR i FROM 3 TO n
    c ‚Üê a + b
    PRINT c

    b ‚Üê c
END FOR
END

9) Palindrome of a Text

Algorithm

START
INPUT str
rev ‚Üê reverse(str)
IF str == rev THEN
    PRINT "Palindrome"
ELSE
    PRINT "Not Palindrome"
END
Note: In some C compilers strrev isn‚Äôt available. A manual reverse is shown below.

Flowchart

Start ‚Üí Read text ‚Üí Reverse text ‚Üí Compare original and reversed ‚Üí Match? ‚Üí Yes ‚Üí Print "Palindrome" : No ‚Üí Print "Not a Palindrome" ‚Üí End

Java Program


import java.util.Scanner;

public class PalindromeCheck {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter text: ");
        String str = sc.nextLine();
        String rev = "";
        
        for (int i = str.length() - 1; i >= 0; i--) {
            rev += str.charAt(i);
        }
        
        if (str.equals(rev)) {
            System.out.println("Palindrome");
        } else {
            System.out.println("Not a Palindrome");
        }
        sc.close();
    }
}

Pseudocode

START
READ text
SET reversed_text ‚Üê REVERSE(text)
IF text = reversed_text THEN
    PRINT "Palindrome"
ELSE
    PRINT "Not a Palindrome"
END IF
END

#include <stdio.h>

10) Check Prime Number

Algorithm

Start
Input n
If n < 2 ‚Üí Not Prime
Check divisors from 2 to sqrt(n)
If any divides ‚Üí Not Prime
Else ‚Üí Prime
End

Flowchart

Start ‚Üí Read n ‚Üí n <= 1? ‚Üí Yes ‚Üí Print "Not Prime" ‚Üí End : No ‚Üí Initialize flag=0 ‚Üí Loop i=2 to ‚àön ‚Üí n MOD i = 0? ‚Üí Yes ‚Üí flag=1 & Break ‚Üí No ‚Üí Next i ‚Üí flag=0? ‚Üí Yes ‚Üí Print "Prime" : No ‚Üí Print "Not Prime" ‚Üí End

Java Program


import java.util.Scanner;

public class PrimeCheck {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number: ");
        int n = sc.nextInt();
        boolean prime = true;

        if (n <= 1) {
            prime = false;
        } else {
            for (int i = 2; i * i <= n; i++) {
                if (n % i == 0) {
                    prime = false;
                    break;
                }
            }
        }

        if (prime) {
            System.out.println("Prime");
        } else {
            System.out.println("Not Prime");
        }
        sc.close();
    }
}

Pseudocode

START
READ n
IF n <= 1 THEN
    PRINT "Not Prime"
ELSE
    SET flag ‚Üê 0
    FOR i ‚Üê 2 TO ‚àön DO
        IF n MOD i = 0 THEN
            SET flag ‚Üê 1
            BREAK
        END IF
    END FOR
    IF flag = 0 THEN
        PRINT "Prime"
    ELSE
        PRINT "Not Prime"
    END IF
END IF
END

    return 0;
}

11) Find Maximum in an Array

Algorithm

START
INPUT n and arr[0..n-1]
max ‚Üê arr[0]
FOR i from 1..n-1:
  IF arr[i] > max THEN max ‚Üê arr[i]
OUTPUT max
END

Flowchart

Start ‚Üí Read n ‚Üí Read array[0..n-1] ‚Üí Set max=array[0] ‚Üí Loop i=1 to n-1 ‚Üí arr[i] > max? ‚Üí Yes ‚Üí max=arr[i] ‚Üí No ‚Üí Next i ‚Üí End of loop ‚Üí Print "Maximum is max" ‚Üí End

Java Program


import java.util.Scanner;

public class MaxInArray {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter array size: ");
        int n = sc.nextInt();
        int[] arr = new int[n];

        System.out.println("Enter elements:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        int max = arr[0];
        for (int i = 1; i < n; i++) {
            if (arr[i] > max) {
                max = arr[i];
            }
        }

        System.out.println("Maximum element is " + max);
        sc.close();
    }
}

START
READ n  // size of array
READ array[1..n]
SET max ‚Üê array[1]
FOR i ‚Üê 2 TO n DO
    IF array[i] > max THEN
        SET max ‚Üê array[i]
    END IF
END FOR
PRINT "Maximum is ", max
END

13) Count Vowels in a String

Algorithm

START
INPUT s
count ‚Üê 0
FOR each ch in s:
  IF ch in {a,e,i,o,u,A,E,I,O,U}:
    count ‚Üê count + 1
OUTPUT count
END

Flowchart

Start ‚Üí Read str ‚Üí Set count=0 ‚Üí For each character ch in str ‚Üí Is ch vowel (a/e/i/o/u)? ‚Üí Yes: count=count+1 / No: continue ‚Üí Next ch ‚Üí Print "Number of vowels" ‚Üí End

Java Program


import java.util.Scanner;

public class CountVowels {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String str = sc.nextLine();
        int count = 0;

        for (int i = 0; i < str.length(); i++) {
            char ch = Character.toLowerCase(str.charAt(i));
            if (ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u') {
                count++;
            }
        }

        System.out.println("Number of vowels: " + count);
        sc.close();
    }
}

Pseudocode (Detailed)

START
READ str
SET count ‚Üê 0
FOR each character ch in str DO
    IF ch is 'a' OR 'e' OR 'i' OR 'o' OR 'u' (ignore case) THEN
        count ‚Üê count + 1
    END IF
END FOR
PRINT "Number of vowels: ", count
END
